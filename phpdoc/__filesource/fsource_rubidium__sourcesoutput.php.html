  45.                     self::$loadedModules[] = self::$mode;
  46.                 }
  47.                 //Set the working module, validate load criteria, get it to return the page 
  48.                 self::$workingModuleName = "module_" . self::$mode;

  53.                 } else {
  54.                     self::load404();
  55.                 }
  56.             } else {
  57.                 self::load404();
  58.             }
  59.         } else {
  60.             //...time for the defaults.
  61.             if(!in_array(rubidium::$settings['default_mode']['value'], self::$loadedModules)) {
  62.                 require (ROOT_PATH . "modules/" . rubidium::$settings['default_mode']['value'] . "/frontend/handler.php");

  64.             }
  65.             self::$workingModuleName = "module_" . rubidium::$settings['default_mode']['value'];
